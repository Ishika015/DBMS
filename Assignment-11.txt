CREATE OR REPLACE PROCEDURE find_minimum (
    num1 IN NUMBER,
    num2 IN NUMBER,
    min_value OUT NUMBER
) IS
BEGIN
    IF num1 < num2 THEN
        min_value := num1;
    ELSE
        min_value := num2;
    END IF;
END;


DECLARE
    num1 NUMBER := 10;
    num2 NUMBER := 20;
    result NUMBER;
BEGIN
    find_minimum(num1, num2, result);
    DBMS_OUTPUT.PUT_LINE('The minimum value is: ' || result);
END;
